int Tree::populateNode(Node treeNode, int enteredData, int dataIndex)
{
	if (treeNode.getLeftPos() != NULL)
	{
		treeNode = treeNode.getLeftPos(); 
		dataIndex = populateNode(treeNode, enteredData, dataIndex); //so it gets updated index
	}
	treeNode.setNodeData(enteredData[dataIndex]);
	dataIndex++;
	if (treeNode.getRightPos() != NULL)
	{
		treeNode = treeNode.getRightPos();
		dataIndex = populateNode(treeNode, enteredData, dataIndex); //so it gets updated index
	}

	return dataIndex;
}
